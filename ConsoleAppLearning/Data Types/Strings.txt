// String Data - same as python, mutiple characters and can be mailipulated
using System;

class Program
{
    static void Main()
    {
        // Declaring and initializing a string
        string message = "Hello, World";

        // Accessing string length
        Console.WriteLine("Messgae: " + message);
        Console.WriteLine("Length of Message: " + message.Length);

        // String concatenation
        string greet = "Good Morning";
        string name = "Jhon";
        string fullMessage = greet + name;

        // Display concatenated string
        Console.WriteLine("Concatenated Message: " + fullMessage);

        // Using string methods
        string upperMessage = message.ToUpper();
        Console.WriteLine("Uppercase Message: " + upperMessage);

        // Substring example
        string subString = message.Substring(7, 5);
        Console.WriteLine("Substring: " + subString);
    }
}
