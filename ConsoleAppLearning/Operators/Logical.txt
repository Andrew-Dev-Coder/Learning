/*
&& - AND
|| - OR
!  - NOT
*/
using System;

class Program
{
    static void Main()
    {
        int a = 6, b = 8;

        Console.WriteLine("a > b && b < 10: " + (a > 5 && b < 10)); // returns true or false based on the end statemeant
        Console.WriteLine("a > 5 || b < 5: " + (a > 5 || b < 5));
        Console.WriteLine("a!(a > 5): " + (!(a > 5)));
    }
}
