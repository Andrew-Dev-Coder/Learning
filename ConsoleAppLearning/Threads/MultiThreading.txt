/*
Mutiple threads execute blocks in parallel
Threads can run independently or interact using synchronization techniques
Methods like Sleep() can pause a thread
*/
using System;
using System.Threading;

class MultiThreadDemo
{
    static void MethodA()
    {
        for (int i = 0; i < 4; i++)
        {
            Console.WriteLine("Method A: {0}", i);
            Thread.Sleep(1000);
        }
    }

    static void MethodB()
    {
        for (int j = 0; j < 4; j++) // ++ adds to j after each loop, for loop checks other conidtions
        {
            Console.WriteLine("MethodB: {0}", j);
            Thread.Sleep(800);
        }
    }

    static void Main()
    {
        Thread t1 = new Thread(MethodA);
        Thread t2 = new Thread(MethodB);

        t1.Start();
        t2.Start();

        t1.Join();
        t2.Join(); // used to synchronize threads

        Console.WriteLine("Both threads completed running");
    }
}
